(define (domain bulb)
	(:Predicates
		(L Лампа горит/свет есть)
		(-L нет света)
		(B Лампа исправна)
		(S Выключатель включен)
		(C Кондиционер включен)
		(W Окно открыто)
	)
	(
		(off_light=Выключить выключатель)
		(on_light=Включить выключатель)
		(ch_bulb=Заменить лампу)
		(off_cond=Выключить кондиционер)
		(on_cond=Включить кондиционер)
		(off_wind=Закрыть окно)
		(on_wind=Открыть окно)
	)
	(: action OffLight 
		:precond (?S)
		:effect (~S)
	)

	(: action OnLight
		:parameters (S L)
		:precond (?S)
		:effect (S & ?L)
	)

	(: action ChBulb :precond (?B & ~S) :effect (B))

	(: action OffCondi :precond (?C) :effect (~C))

	(: action OnCondi :precond (?C) :effect (C))

	(: action OffWind :precond (?W) :effect (~W))

	(: action OnWind :precond (?W) :effect (W))

	(: axiom (L <-> B & S))
)

(define (problem switch_on)
	(:domain bulb)
	(:init (~L))
	(:goal (L)))
