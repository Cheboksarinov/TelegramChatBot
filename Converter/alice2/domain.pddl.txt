(define (domain bulb)
	(:Predicates
		(Light Лампа горит/свет есть)
		(-Light нет света)
		(Bulb Лампа исправна)
		(Switch Выключатель включен)
		(Condi Кондиционер включен)
		(Window Окно открыто)
	)
	(
		(off_light=Выключить выключатель)
		(on_light=Включить выключатель)
		(ch_bulb=Заменить лампу)
		(off_cond=Выключить кондиционер)
		(on_cond=Включить кондиционер)
		(off_wind=Закрыть окно)
		(on_wind=Открыть окно)
	)
	(: action OffLight 
		:precond (?Switch)
		:effect (~Switch)
	)

	(: action OnLight
		:parameters (Switch Light)
		:precond (?Switch)
		:effect (Switch & ?Light)
	)

	(: action ChBulb :precond (?Bulb & ~Switch) :effect (Bulb))

	(: action OffCondi :precond (?Condi) :effect (~Condi))

	(: action OnCondi :precond (?Condi) :effect (Condi))

	(: action OffWind :precond (?Window) :effect (~Window))

	(: action OnWind :precond (?Window) :effect (Window))

	(: axiom (Light <-> Bulb & Switch))
)

(define (problem switch_on)
	(:domain bulb)
	(:init (~Light))
	(:goal (Light)))
